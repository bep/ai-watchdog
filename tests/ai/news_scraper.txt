+import requests
+from bs4 import BeautifulSoup
+from fastapi import FastAPI, HTTPException
+from typing import List, Dict
+import re
+from datetime import datetime
+
+app = FastAPI()
+
+class ABCNewsScraper:
+    def __init__(self):
+        self.base_url = "https://abcnews.go.com/search/results?searchtext="
+        self.headers = {
+            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.124 Safari/537.36"
+        }
+
+    def scrape_news(self, topic: str) -> List[Dict]:
+        try:
+            url = f"{self.base_url}{topic}"
+            response = requests.get(url, headers=self.headers)
+            response.raise_for_status()
+            
+            soup = BeautifulSoup(response.content, 'html.parser')
+            articles = soup.find_all('div', class_='result-item')
+            
+            news_items = []
+            for article in articles:
+                try:
+                    title = article.find('h3').text.strip()
+                    link = article.find('a')['href']
+                    date_element = article.find('span', class_='date')
+                    date = date_element.text.strip() if date_element else "No date available"
+                    description = article.find('div', class_='desc').text.strip()
+                    
+                    news_items.append({
+                        "title": title,
+                        "link": link,
+                        "date": date,
+                        "description": description
+                    })
+                except AttributeError:
+                    continue
+                    
+            return news_items
+        except Exception as e:
+            raise HTTPException(status_code=500, detail=f"Failed to scrape news: {str(e)}")
+
+scraper = ABCNewsScraper()
+
+@app.get("/news/{topic}")
+async def get_news_by_topic(topic: str):
+    """
+    Get recent news articles from ABC News based on the provided topic.
+    
+    Args:
+        topic (str): The topic to search for
+        
+    Returns:
+        List[Dict]: A list of news articles containing title, link, date, and description
+    """
+    try:
+        news_items = scraper.scrape_news(topic)
+        if not news_items:
+            raise HTTPException(status_code=404, detail=f"No news found for topic: {topic}")
+        return news_items
+    except Exception as e:
+        raise HTTPException(status_code=500, detail=str(e))
+
+if __name__ == "__main__":
+    import uvicorn
+    uvicorn.run(app, host="0.0.0.0", port=8000)
